Put in this order:

Role
CREATE TABLE IF NOT EXISTS role(id INTEGER PRIMARY KEY, role_name VARCHAR(255) NOT NULL);

School
CREATE TABLE IF NOT EXISTS school(id INTEGER PRIMARY KEY, school VARCHAR(255) NOT NULL);

Class
CREATE TABLE IF NOT EXISTS class(id INTEGER PRIMARY KEY, class_num INTEGER NOT NULL, class_name VARCHAR(255) NOT NULL, class_desc VARCHAR(255));

Now the other tables with foreign keys can be put in:
User
CREATE TABLE IF NOT EXISTS user(id INTEGER PRIMARY KEY, password VARCHAR(128) NOT NULL, email VARCHAR(255) NOT NULL, name VARCHAR(255) NOT NULL, username VARCHAR(255) NOT NULL, roleId INTEGER NOT NULL, isProfessor BOOLEAN, schoolId INTEGER NOT NULL, FOREIGN KEY(roleId) REFERENCES role(id), FOREIGN KEY(schoolId) REFERENCES school(id));

Posts
CREATE TABLE IF NOT EXISTS posts(id INTEGER PRIMARY KEY, authorId INTEGER NOT NULL, classId INTEGER NOT NULL, text TEXT NOT NULL, rating FLOAT, endorse BOOLEAN, FOREIGN KEY(authorId) REFERENCES user(id), FOREIGN KEY(classId) REFERENCES class(id));

Comments
CREATE TABLE IF NOT EXISTS comments(id INTEGER PRIMARY KEY, postId INTEGER NOT NULL, comment TEXT NOT NULL, authorId INTEGER NOT NULL, score INTEGER, FOREIGN KEY(postId) REFERENCES posts(id), FOREIGN KEY(authorId) REFERENCES user(id));

Image
CREATE TABLE IF NOT EXISTS image(id INTEGER PRIMARY KEY, fileLocation VARCHAR(255) NOT NULL, postId INTEGER NOT NULL, FOREIGN KEY(postId) REFERENCES posts(id));
